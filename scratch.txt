class ASuper {
  x(thing: B<number>) { ... }
}

class A<T> extends ASuper {
  @val: T

  constructor(T val) {
    @val = val
  }

  x(thing: B<int>): T {
    return @val
  }
}


class BSuper {
  x(thing: A<number>) { ... }
}

class B<T> extends BSuper {
  @val: T

  constructor(T val) {
    @val = val
  }

  x(thing: A<int>): T {
    return @val
  }
}
